```python
def optimized_power(x, n):
	"""Compute the value x**n for integer n"""
	if n == 0:
		return 1
	else:
		partial = optimized_power(x, n // 2)
		result = partial * partial
		
		if n % 2 == 1:
			result *= x
	return result
```

$$power(n, x) =
  \begin{cases}
    1       & \quad \text{if } n = 0 \\
    x*(power(x, \lfloor n/2 \rfloor))^2  & \quad \text{if } n \text{ is odd} \\
    (power(x, \lfloor n/2 \rfloor))^2 & \quad \text{if } n \text{ is even}
  \end{cases}
