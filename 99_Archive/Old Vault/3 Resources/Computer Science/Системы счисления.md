# Алгоритм перевода целых чисел

```pseudocode
1. Input n
2. Repeat
	2.1 Output n mod base
	2.2 n = n div base
	until n = 0
3. Read outputs in reverse order
```
Пример кода на [[Swift Foundation]]:
```swift
func convertToBase(number: Int, to base: Int) -> String {
    func toHex(_ number: Int) -> String {
        switch number {
        case 10: return "A"
        case 11: return "B"
        case 12: return "C"
        case 13: return "D"
        case 14: return "E"
        case 15: return "F"
        default: return String(number)
        }
    }
    
    var t = number
    var i = ""
    repeat {
        i += toHex(t % base)
        t = t / base
        
    } while t > 0
    return String(i.reversed())
}
```

# Алгоритм перевода дробных чисел
```pseudo
1. Input n and d // n - fractional number, d - digits
2. i = 0
3. Repeat
	3.1 i = i + 1
	3.2 m = base * n
	3.3 Output floor(m)
	3.4 n = frac(m)
	until n = 0 or i = d
4. Read outputs
```
Пример кода:
```swift
func convertFractional(number: Float, to base: Int, with precision: Int) -> String {
    func toHex(_ number: Int) -> String {
        switch number {
        case 10: return "A"
        case 11: return "B"
        case 12: return "C"
        case 13: return "D"
        case 14: return "E"
        case 15: return "F"
        default: return String(number)
        }
    }
    var result = ""
    
    var integral = Int(number)
    var fractional = number - Float(integral)
    var k = precision
    
    while integral > 0 {
        let remainder = integral % base
        result += toHex(remainder)
        integral /= base
    }
    
    result = String(result.reversed())
    result += "."
    
    while k > 0 {
        fractional *= Float(base)
        let integral = floor(fractional)
        result += toHex(Int(integral))
        fractional -= integral
        k -= 1
    }
    
    return result
}
```
