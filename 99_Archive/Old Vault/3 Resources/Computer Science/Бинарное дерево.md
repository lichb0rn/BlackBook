- Двоичное дерево - [[Algorithms|структура данных]], в которой каждый узел имеет не более двух потомков.

Обход дерева можно выполнить тремя вариантами:
- *In-Order*: сначала спускаемся до конца по левому пути, потом обходим себя, потом идём направо.
- *Pre-Order*: сначала заходим в себя, потом уходим налево, затем направо.
- *Post-Order*: сначала лево, потом право, в конце себя.

Все операции обхода - **$O(n)$**

## Удаление элемента
Удаление узла можно разделить на 3 случая:
- **Удаления листа**. В этом случае мы просто удаляем элемент, структура дерева не нарушается.
![[Pasted image 20221020125153.png]]
- **Удаление узла с одним потомком**. Узел удаляем, а его родителя соединяем с потомками:
![[Pasted image 20221020125234.png]]
- **Удаление узла с двумя потомками**. В этом случае нам нужно найти *приемника*, чтобы сохранить структуру дерева. Приемника - следующий узел, если бы мы обходили дерево в отсортированном порядке. 
Мы помещаем приемника на место удаленного узла.  Чтобы найти приемника, мы можем пойти по правому поддереву. Тогда приемника будет самым левым узлом (минимальным) в этом поддереве.

## Реализиация
- Реализация на [[Swift]]:
```swift
public class BinaryNode<Element> {
    
    public var value: Element
    public var leftChild: BinaryNode?
    public var rightChild: BinaryNode?
    
    public init(value: Element) {
        self.value = value
    }
}

extension BinaryNode {
    
    public func traverseInOrder(visit: (Element) -> Void) {
        leftChild?.traverseInOrder(visit: visit)
        visit(value)
        rightChild?.traverseInOrder(visit: visit)
    }
    
    public func traversePreOrder(visit: (Element) -> Void) {
        visit(value)
        leftChild?.traversePreOrder(visit: visit)
        rightChild?.traversePreOrder(visit: visit)
    }
    
    public func traversePostOrder(visit: (Element) -> Void) {
        leftChild?.traversePostOrder(visit: visit)
        rightChild?.traversePostOrder(visit: visit)
        visit(value)
    }
}

extension BinaryNode: CustomStringConvertible {
    public var description: String {
        return diagram(for: self)
    }
    
    private func diagram(for node: BinaryNode?,
                         _ top: String = "",
                         _ root: String = "",
                         _ bottom: String = "") -> String {
        
        guard let node = node else {
            return root + "nil\n"
        }
        
        if node.leftChild == nil && node.rightChild == nil {
            return root + "\(node.value)\n"
        }
        
        return diagram(for: node.rightChild,
                          top + " ", top + "┌──", top + "│ ")
                + root + "\(node.value)\n"
                + diagram(for: node.leftChild,
                             bottom + "│ ", bottom + "└──", bottom + " ")
    }
}```

- Реализация на  [[Python]]:
```python
class TreeNode:  
    def __init__(self, value, left = None, right = None, parent = None):  
        self.value = value  
        self.left = left  
        self.right = right  
        self.parent = parent  
  
    def __str__(self):  
        return f"{self.value}"  
  
    def __repr__(self) -> str:  
        return f"{self.value}"
    
```

```python
class BinaryTree:  
    def __init__(self, root=None) -> None:  
        self.root = root  
  
    def print_tree(self, node, level=0):  
        if node is not None:  
            self.print_tree(node.left, level + 1)  
            print(' ' * 4 * level + '-> ' + str(node.value))  
            self.print_tree(node.right, level + 1)  
  
    def find_value(self, target):  
        return self._find_value(target, self.root)  
  
    def _find_value(self, target, current):  
        if current is None:  
            return None  
        if current.value == target:  
            return current  
        elif target < current.value and current.left is not None:  
            return self._find_value(target, current.left)  
        elif target > current.value and current.right is not None:  
            return self._find_value(target, current.right)  
        else:  
            return None  
  
    def insert(self, new_value):  
        if self.root is None:  
            self.root = TreeNode(new_value)  
        else:  
            self._insert(self.root, new_value)  
  
    def _insert(self, current, new_value):  
        if new_value == current.value:  
            return  
        if new_value < current.value:  
            if current.left is not None:  
                self._insert(current.left, new_value)  
            else:  
                current.left = TreeNode(new_value)  
                current.left.parent = current  
        else:  
            if current.right is not None:  
                self._insert(current.right, new_value)  
            else:  
                current.right = TreeNode(new_value)  
                current.right.parent = current  
  
    def remove(self, node):  
        if self.root is None or node is None:  
            return  
        self._remove(self.root, node)  
  
    def _remove(self, current, node):  
        # Deleting a leaf node  
        if node.left is None and node.right is None:  
            if node.parent is None:  
                self.root = None  
            elif node.parent.left == node:  
                node.parent.left = None  
            else:  
                node.parent.right = None  
            return  
        # Deleting node with one child  
        if node.left is None or node.right is None:  
            child = node.left  
            if node.left is None:  
                child = node.right  
  
            child.parent = node.parent  
            if node.parent is None:  
                self.root = child  
            elif node.parent.left == node:  
                node.parent.left = child  
            else:  
                node.parent.right = child  
            return  
  
        # Deleting a node with 2 children  
        # Find the successor and splice it out of the tree        successor = node.right  
        while successor.left is not None:  
            successor = successor.left  
        self.remove(successor)  
  
        # Insert the successor in the deleted node's place  
        if node.parent is None:  
            self.root = successor  
        elif node.parent.left == node:  
            node.parent.left = successor  
        else:  
            node.parent.right = successor  
        successor.parent = node.parent  
  
        successor.left = node.left  
        node.left.parent = successor  
  
        successor.right = node.right  
        if node.right is not None:  
            node.right.parent = successor  
  
        node.parent = None  
        node.left = None  
        node.right = None
