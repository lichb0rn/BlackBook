public func generateMaze(rows: Int, columns: Int, sparseness: Double) -> Maze {
    // init an empty maze
    var maze: Maze = Maze(repeating: [Cell](repeating: .Empty, count: columns),
                          count: rows)

        }
    }
    
    return maze
}
