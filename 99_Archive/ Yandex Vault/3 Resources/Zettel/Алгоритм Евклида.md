function gcd(x, y) {
	larger = Math.max(x, y);
	smaller = Math.min(x, y);

	remainder = larger % smaller;

	if (remainder === 0) {
		return smaller;
	}

	if (remainder !== 0) {
		return gcd(smaller, remainder);
	}
}
```

def gcd(x ,y):
    """Euclidean algorithm for finding gcd"""
    larger = max(x, y)
    smaller = min(x, y)

    remainder = larger % smaller
    if remainder == 0:
        return smaller

    if remainder != 0:
        return gcd(smaller, remainder)
```
