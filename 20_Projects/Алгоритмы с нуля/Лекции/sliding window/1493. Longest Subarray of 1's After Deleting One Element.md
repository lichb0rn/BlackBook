## Итерация 1

```go
func longestSubarray(nums []int) int {
    k := 1
    begin := 0
    // сколько 0 мы видели
    windowState := 0
    result := 0

    for end := range len(nums) {
       if nums[end] == 0 {
            windowState++
       }
    
        // сжимаем, когда кол-во нулей > k 
        for windowState > k {
            if nums[begin] == 0 {
                windowState--
            }

            begin++ // shrink window
        }

        result = max(result, end - begin + 1)
    }

    return result - 1 
}

func max(a, b int) int {
	if a > b {
		return a
	}
	return b
}
```
