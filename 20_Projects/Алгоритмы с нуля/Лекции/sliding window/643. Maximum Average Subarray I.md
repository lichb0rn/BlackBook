## Итерация 1
```go
func findMaxAverage(nums []int, k int) float64 {
    begin := 0
    // сумма всех элементов в рамке
    var windowState float64 = 0
    result := math.Inf(-1)

    for end := range len(nums) {
        windowState += float64(nums[end])

        windowSize := end - begin + 1
        if windowSize == k {
            result = math.Max(result, windowState)

            windowState -= float64(nums[begin])
            begin++
        }
    }
    return result / float64(k)    
}
```
