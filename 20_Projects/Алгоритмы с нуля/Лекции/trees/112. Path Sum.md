/**
 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val int
 *     Left *TreeNode
 *     Right *TreeNode
 * }
 */
func hasPathSum(root *TreeNode, targetSum int) bool {
    if root == nil {
        return false
    }

    stack := []Step{Step{root, 0}}
    for len(stack) > 0 {
        step := stack[len(stack)-1]
		stack = stack[:len(stack)-1]

        node := step.Node
        sum := step.Sum

        if node == nil {
            continue
        }

        sum += node.Val

        if node.Left == nil && node.Right == nil && sum == targetSum {
            return true
        }

        stack = append(stack, Step{node.Left, sum})
        stack = append(stack, Step{node.Right, sum})
    }

    return false
}

type Step struct {
    Node *TreeNode
    Sum int
}
