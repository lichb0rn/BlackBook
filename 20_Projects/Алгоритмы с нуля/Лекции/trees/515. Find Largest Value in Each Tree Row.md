/**
 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val int
 *     Left *TreeNode
 *     Right *TreeNode
 * }
 */
func largestValues(root *TreeNode) []int {
 	if root == nil {
		return []int{}
	}

	result := make([]int, 0)
	q := Queue{}
	q.Enqueue(root)
	for !q.IsEmpty() {
		levelSize := q.Size()
        levelMax := math.MinInt

		for i := 0; i < levelSize; i++ {
			node := q.Dequeue()
			levelMax = max([]int{levelMax, node.Val})

			if node.Left != nil {
				q.Enqueue(node.Left)
			}
			if node.Right != nil {
				q.Enqueue(node.Right)
			}
		}

		result = append(result, levelMax)
	}

	return result   
}

func max(s []int) int {
    m := s[0]
    for _, v := range s {
        if v > m {
            m = v
        }
    }
    return m
}

type Queue []*TreeNode

// Enqueue adds an element to the rear of the queue
func (q *Queue) Enqueue(value *TreeNode) {
	*q = append(*q, value)
}

// Dequeue removes and returns an element from the front of the queue
func (q *Queue) Dequeue() *TreeNode {
	if q.IsEmpty() {
		return nil
	}
	value := (*q)[0]
	*q = (*q)[1:]
	return value
}

// IsEmpty checks if the queue is empty
func (q *Queue) IsEmpty() bool {
	return len(*q) == 0
}

// Size returns the number of elements in the queue
func (q *Queue) Size() int {
	return len(*q)
}
