/**
 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val int
 *     Left *TreeNode
 *     Right *TreeNode
 * }
 */
func isValidBST(root *TreeNode) bool {
    stack := []Step{{root, math.MinInt, math.MaxInt}}

    for len(stack) > 0 {
        step := stack[len(stack)-1]
		stack = stack[:len(stack)-1]

        node := step.Node
        lower := step.MinVal
        upper := step.MaxVal

        if node == nil {
            continue
        }

        if node.Val <= lower || node.Val >= upper {
            return false
        }

        stack = append(stack, Step{node.Left, lower, node.Val})
        stack = append(stack, Step{node.Right, node.Val, upper})
    }
    return true
}

type Step struct {
    Node *TreeNode
    MinVal int
    MaxVal int
}
