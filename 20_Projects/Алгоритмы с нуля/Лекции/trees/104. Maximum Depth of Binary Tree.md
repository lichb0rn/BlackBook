## Рекурсивный вариант

```go
/**
 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val int
 *     Left *TreeNode
 *     Right *TreeNode
 * }
 */
func maxDepth(root *TreeNode) int {
    if root == nil {
        return 0
    }
    l := maxDepth(root.Left)
    r := maxDepth(root.Right)
    return max(l, r) + 1

}

func max(a, b int) int {
    if a > b {
        return a
    } else {
        return b
    }
}
```

## Итеративный вариант
```go
/**
 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val int
 *     Left *TreeNode
 *     Right *TreeNode
 * }
 */
func maxDepth(root *TreeNode) int {
	if root == nil {
		return 0
	}

	stack := []Step{{root, 1}}
	result := 0

	for len(stack) > 0 {
		step := stack[len(stack)-1]
		stack = stack[:len(stack)-1]

		node := step.Node
		depth := step.Depth

		if node == nil {
			continue
		}

		result = max(result, depth)
		stack = append(stack, Step{node.Left, depth + 1})
		stack = append(stack, Step{node.Right, depth + 1})
	}

	return result
}
func max(a, b int) int {
	if a > b {
		return a
	} else {
		return b
	}
}

type Step struct {
	Node  *TreeNode
	Depth int
}
```
