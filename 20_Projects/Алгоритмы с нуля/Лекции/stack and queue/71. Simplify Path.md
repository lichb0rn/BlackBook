func simplifyPath(path string) string {
	splited := strings.Split(path, "/")
	st := NewStack(len(splited))
	for _, p := range splited {
		if p == "" || p == "." {
			continue
		}
		if p == ".." {
			st.Pop()
		} else {
			st.Push(p)
		}
	}

	var result string
	for i := 0; i < st.Size(); i++ {
		result += "/" + st.arr[i]
	}

	if result == "" {
		return "/"
	}

	return result
}

type stack struct {
	arr []string
}

func NewStack(l int) stack {
	return stack{
		arr: make([]string, 0, l),
	}
}

func (s *stack) Push(e string) {
	s.arr = append(s.arr, e)
}

func (s *stack) Pop() (string, bool) {
	if s.Size() == 0 {
		return "", false
	}
	prev := s.arr[len(s.arr)-1]
	s.arr = s.arr[:len(s.arr)-1]
	return prev, true
}

func (s *stack) Size() int {
	return len(s.arr)
}

func (s *stack) Peek() (string, bool) {
	if s.Size() == 0 {
		return "", false
	}
	return s.arr[len(s.arr)-1], true
}
