func isValid(s string) bool {
	pairs := map[rune]rune{
		'(': ')',
		'{': '}',
		'[': ']',
	}

	l := utf8.RuneCountInString(s)
	ss := NewStack(l)

	for _, c := range s {
		_, ok := pairs[c]

		if ok {
			ss.Push(c)
		} else {
			if ss.Size() == 0 {
				return false
			}
			prev := ss.Pop()
			obr, ok := pairs[prev]
			if !ok || obr != c {
				return false
			}
		}
	}

	return ss.Size() == 0
}

type stack struct {
	arr []rune
}

func NewStack(l int) stack {
	return stack{
		arr: make([]rune, 0, l),
	}
}

func (s *stack) Push(e rune) {
	s.arr = append(s.arr, e)
}

func (s *stack) Pop() rune {
	prev := s.arr[len(s.arr)-1]
	s.arr = s.arr[:len(s.arr)-1]
	return prev
}

func (s *stack) Size() int {
	return len(s.arr)
}
